### Meta API Time Range

```typescript
// Atual: date_preset=last_30d
// Novo:  time_range={"since":"2026-02-01","until":"2026-03-01"}

function getTimeParam(days?: number, startDate?: string, endDate?: string): string {
  if (startDate && endDate) {
    return `time_range=${encodeURIComponent(JSON.stringify({ since: startDate, until: endDate }))}`;
  }
  const datePreset = days <= 7 ? "last_7d" : days <= 14 ? "last_14d" : days <= 30 ? "last_30d" : "last_90d";
  return `date_preset=${datePreset}`;
}
```

### CSV Export

```typescript
function exportCsv(campaigns: CampaignAnalytics[], projectName: string, startDate: string, endDate: string) {
  const headers = ["Nome", "Spend", "Impressoes", "Cliques", "CTR", "CPC", "CPM", ...conditionalHeaders];
  const rows = campaigns.map(c => [
    c.campaignName,
    c.spend.toFixed(2).replace(".", ","),
    c.impressions,
    c.clicks,
    c.ctr.toFixed(2).replace(".", ","),
    // ...
  ]);

  const csv = [headers.join(";"), ...rows.map(r => r.join(";"))].join("\n");
  const blob = new Blob(["\ufeff" + csv], { type: "text/csv;charset=utf-8;" }); // BOM for Excel
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = `loyola-traffic-${projectName}-${startDate}-${endDate}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}
```
