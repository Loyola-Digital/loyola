## Story

**As a** gestor de trafego da Loyola Digital,
**I want** to configure a qualified lead profile per project (e.g., female + income >5k + catholic + up to 2 children) and have the system automatically classify leads from the survey spreadsheet,
**so that** I can see how many qualified leads each campaign generated and calculate cost per qualified lead.

### Nova Tabela — qualification_profiles

```typescript
export const qualificationProfiles = pgTable(
  "qualification_profiles",
  {
    id: uuid("id").defaultRandom().primaryKey(),
    projectId: uuid("project_id")
      .notNull()
      .references(() => projects.id, { onDelete: "cascade" }),
    rules: jsonb("rules")
      .$type<QualificationRule[]>()
      .notNull(),
    createdBy: uuid("created_by")
      .notNull()
      .references(() => users.id, { onDelete: "restrict" }),
    createdAt: timestamp("created_at", { withTimezone: true })
      .defaultNow()
      .notNull(),
    updatedAt: timestamp("updated_at", { withTimezone: true })
      .defaultNow()
      .notNull(),
  },
  (table) => [
    uniqueIndex("uq_qualification_project").on(table.projectId),
  ]
);

// Rule type:
interface QualificationRule {
  field: string;       // nome do campo da pesquisa (ex: "sexo", "renda", "filhos", "religiao")
  operator: "equals" | "not_equals" | "gte" | "lte" | "contains" | "in";
  value: string;       // valor para comparar (para "in": valores separados por virgula)
}
```

### Qualification Logic

```typescript
// lead-qualification.ts

function isLeadQualified(
  surveyData: Record<string, string>,  // dados da pesquisa deste lead
  rules: QualificationRule[]
): boolean {
  // AND logic: ALL rules must pass
  return rules.every((rule) => {
    const fieldValue = (surveyData[rule.field] ?? "").trim().toLowerCase();
    const ruleValue = rule.value.trim().toLowerCase();

    switch (rule.operator) {
      case "equals":     return fieldValue === ruleValue;
      case "not_equals": return fieldValue !== ruleValue;
      case "gte":        return parseFloat(fieldValue) >= parseFloat(ruleValue);
      case "lte":        return parseFloat(fieldValue) <= parseFloat(ruleValue);
      case "contains":   return fieldValue.includes(ruleValue);
      case "in":         return ruleValue.split(",").map(v => v.trim()).includes(fieldValue);
      default:           return false;
    }
  });
}
```
